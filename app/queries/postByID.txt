query post($id: Int!) {
  post(id: $id) {
    id
    title
    preview(removeMarkdown: true, length: 150)
    body
    isAnnouncement
    url
    isAnswerable
    isHidden
    answeredBy {
      id
      ...PostAnsweredCardUser
    }
    answer {
      id
      ...PostAnsweredCardComment
    }
    board {
      id
      url
      description
      slug
    }
    ...PostDetailPost
  }
}
fragment PostDetailPost on Post {
  id
  title
  body
  timeCreated
  canReport
  hasReported
  isAnnouncement
  isPinned
  isLocked
  isHidden
  commentCount
  isAnswered
  isAnswerable
  ...PostVoteControlPost
  user {
    id
    timeCreated
    ...UserPostHeaderUser
  }
  repl {
    id
    url
    title
    description
    isPrivate
    isAlwaysOn
    lang {
      id
      icon
      key
      displayName
      tagline
    }
  }
  board {
    id
    name
    color
  }
}
fragment PostVoteControlPost on Post {
  id
  voteCount
  canVote
  hasVoted
}
fragment UserPostHeaderUser on User {
  id
  image
  isHacker
  isModerator: hasRole(role: MODERATOR)
  isAdmin: hasRole(role: ADMIN)
  ...DepreciatedUserLabelUser
  ...UserLinkUser
}
fragment UserLinkUser on User {
  id
  url
  username
}
fragment DepreciatedUserLabelUser on User {
  id
  image
  username
  url
  karma
}
fragment PostAnsweredCardUser on User {
  id
  username
  ...DepreciatedUserLabelUser
}
fragment PostAnsweredCardComment on Comment {
  id
  url
}
