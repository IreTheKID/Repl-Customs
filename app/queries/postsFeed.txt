query PostsFeed(
  $order: String
  $after: String
  $searchQuery: String
  $languages: [String!]
  $count: Int
  $boardSlugs: [String!]
  $pinAnnouncements: Boolean
  $pinPinned: Boolean
) {
  posts(
    order: $order
    after: $after
    searchQuery: $searchQuery
    languages: $languages
    count: $count
    boardSlugs: $boardSlugs
    pinAnnouncements: $pinAnnouncements
    pinPinned: $pinPinned
  ) {
    pageInfo {
      nextCursor
    }
    items {
      id
      timeCreated
      ...PostsFeedItemPost
      board {
        id
        color
        name
      }
    }
  }
}
fragment PostsFeedItemPost on Post {
  id
  title
  preview(removeMarkdown: true, length: 150)
  url
  commentCount
  isPinned
  isLocked
  isHidden
  isAnnouncement
  timeCreated
  isAnswered
  isAnswerable
  ...PostVoteControlPost
  ...PostLinkPost
  user {
    id
    fullName
    username
    image
    bio
    karma
    isHacker
    timeCreated
  }
  repl {
    id
    url
    title
    description
    isPrivate
    isAlwaysOn
    lang {
      id
      icon
      key
      displayName
      tagline
    }
  }
  board {
    id
    name
    color
  }
}
fragment PostVoteControlPost on Post {
  id
  voteCount
  canVote
  hasVoted
}
fragment PostLinkPost on Post {
  id
  url
}
